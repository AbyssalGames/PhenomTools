using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

[Serializable]
public class #SCRIPTNAME#_Stats : Weapon_MeleeStatsBase
{

}

public class #SCRIPTNAME# : Weapon_MeleeBase
{
    [Space]
    [SerializeField]
    private #SCRIPTNAME#_Stats baseStats;

    public override Weapon_MeleeStatsBase weaponMeleeStats => stats;
    public virtual #SCRIPTNAME#_Stats stats { get; private set; }

    public override void OnCreated(ItemObject item)
    {
        stats = baseStats;
        stats.objectInfo = item;

        base.OnCreated(item);
        RollStats();
    }

    public override void OnAttack()
    {
        if (Time.time > nextAttackTime)
        {
            nextAttackTime = Time.time + 1 / stats.attackSpeed;
            base.OnAttack();
        }
    }

    public override void RollStats()
    {
        base.RollStats();
    }
}
